-- Script do Banco de Dados Housing


-- Criando Schema
begin;
drop schema if exists housing; 
create schema housing;
use housing;

-- Criando as tabelas

-- Tabela REGIONSTATE
CREATE TABLE REGIONSTATE (
    region VARCHAR(31),
    state TEXT,
    PRIMARY KEY (region)
);

-- Tabela LATLONGREGION
CREATE TABLE LATLONGREGION(
latitude DOUBLE SIGNED ,
longitude DOUBLE SIGNED ,
region VARCHAR(31),
PRIMARY KEY(latitude, longitude),
FOREIGN KEY(region) REFERENCES REGIONSTATE(region)
);

CREATE TABLE PRINCIPAL(
id BIGINT,
region VARCHAR(31),
price DOUBLE,
houseType TEXT,
sqFeet INT,
beds INT,
baths INT,
catsAllowed INT,
dogsAllowed INT,
smokingAllowed INT,
comesFurnished INT,
latitude DOUBLE SIGNED,
longitude DOUBLE SIGNED,
PRIMARY KEY(id),
FOREIGN KEY(region) REFERENCES REGIONSTATE(region)
);
